-- Lưu config
local config = _G.KaitunV99

-- ===== CORE FUNCTIONS =====
local function getPlayer()
    return game.Players.LocalPlayer
end

local function getCharacter()
    return getPlayer().Character or getPlayer().CharacterAdded:Wait()
end

local function getHumanoid()
    return getCharacter():WaitForChild("Humanoid")
end

local function getRoot()
    return getCharacter():WaitForChild("HumanoidRootPart")
end

local function getLevel()
    return getPlayer():GetAttribute("Level") or 0
end

local function getBeli()
    return getPlayer():GetAttribute("Beli") or 0
end

local function getFrags()
    return getPlayer():GetAttribute("Fragments") or 0
end

-- ===== FIND TARGET =====
local function findNearestMonster()
    local root = getRoot()
    if not root then return nil end
    local nearest = nil
    local minDist = math.huge
    
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Model") and v:FindFirstChild("Humanoid") and v:FindFirstChild("HumanoidRootPart") then
            if v.Name ~= getPlayer().Name and v.Humanoid.Health > 0 then
                local dist = (root.Position - v.HumanoidRootPart.Position).Magnitude
                if dist < minDist and dist < 300 then
                    nearest = v
                    minDist = dist
                end
            end
        end
    end
    return nearest
end

local function findNearestBoss()
    local root = getRoot()
    if not root then return nil end
    local nearest = nil
    local minDist = math.huge
    
    for _, bossName in pairs(config.BossList) do
        for _, v in pairs(game.Workspace:GetChildren()) do
            if v:IsA("Model") and v.Name:find(bossName) then
                local hrp = v:FindFirstChild("HumanoidRootPart")
                local hum = v:FindFirstChild("Humanoid")
                if hrp and hum and hum.Health > 0 then
                    local dist = (root.Position - hrp.Position).Magnitude
                    if dist < minDist and dist < 500 then
                        nearest = v
                        minDist = dist
                    end
                end
            end
        end
    end
    return nearest
end

-- ===== FAST ATTACK =====
local function fastAttack(target)
    if not target or not config.FastAttack then return end
    
    local humanoid = getHumanoid()
    local root = getRoot()
    local targetRoot = target:FindFirstChild("HumanoidRootPart")
    
    if not targetRoot then return end
    
    -- Teleport
    if config.TeleportToMonsters then
        root.CFrame = targetRoot.CFrame * CFrame.new(0, 0, 4)
        wait(0.01)
    end
    
    -- Spam M1
    for i = 1, 10 do
        humanoid:Attack()
        wait(config.AttackSpeed)
    end
    
    -- Spam Skills
    if config.SpamSkills then
        for _, tool in pairs(getCharacter():GetChildren()) do
            if tool:IsA("Tool") then
                local remote = tool:FindFirstChild("RemoteEvent")
                if remote then
                    for _, key in pairs({"Z", "X", "C", "V", "F", "G", "H"}) do
                        remote:FireServer(key)
                        wait(config.SkillDelay)
                    end
                end
            end
        end
        
        -- Remote skill
        pcall(function()
            local remote = game:GetService("ReplicatedStorage"):FindFirstChild("Remotes"):FindFirstChild("Skill")
            if remote then
                for _, key in pairs({"Z", "X", "C", "V"}) do
                    remote:FireServer(key)
                    wait(0.01)
                end
            end
        end)
    end
end

-- ===== AUTO HEAL =====
local function autoHeal()
    local humanoid = getHumanoid()
    if not humanoid then return end
    
    local hpPercent = (humanoid.Health / humanoid.MaxHealth) * 100
    if hpPercent < config.HealThreshold then
        local player = getPlayer()
        for _, item in pairs(player.Backpack:GetChildren()) do
            if item:IsA("Tool") then
                local name = string.lower(item.Name)
                if name:find("heal") or name:find("potion") or name:find("fruit") or name:find("food") then
                    humanoid:EquipTool(item)
                    wait(0.1)
                    item:Activate()
                    wait(0.5)
                    if (humanoid.Health / humanoid.MaxHealth) * 100 > 70 then break end
                end
            end
        end
    end
end

-- ===== AUTO COLLECT =====
local function autoCollect()
    if not config.AutoCollect then return end
    
    local root = getRoot()
    if not root then return end
    
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Tool") or v:IsA("Model") then
            local name = string.lower(v.Name)
            if name:find("drop") or name:find("fruit") or name:find("chest") or name:find("beli") or name:find("frag") then
                if v:FindFirstChild("ClickDetector") then
                    local dist = (root.Position - v.Position).Magnitude
                    if dist < config.CollectRadius then
                        fireclickdetector(v:FindFirstChild("ClickDetector"))
                        wait(0.1)
                    end
                end
            end
        end
    end
end

-- ===== AUTO QUEST =====
local function autoQuest()
    if not config.AutoQuest then return end
    
    local player = getPlayer()
    local character = getCharacter()
    if not character then return end
    
    -- Check quest
    for _, v in pairs(game.Workspace:GetChildren()) do
        if v:IsA("Model") and v:FindFirstChild("QuestGiver") then
            local dist = (getRoot().Position - v.HumanoidRootPart.Position).Magnitude
            if dist < 30 then
                -- Accept/Turn in quest
                local remote = v:FindFirstChild("RemoteEvent")
                if remote then
                    remote:FireServer()
                    wait(0.5)
                end
            end
        end
    end
end

-- ===== SERVER HOP =====
local function serverHop()
    if not config.ServerHop then return end
    
    local servers = game:GetService("TeleportService"):GetServerList()
    if #servers > 1 then
        local target = servers[math.random(2, #servers)]
        game:GetService("TeleportService"):TeleportToServer(game.PlaceId, target)
        wait(3)
    end
end

-- ===== MAIN LOOP =====
coroutine.wrap(function()
    while wait(0.05) do
        if not config.AutoFarm then 
            wait(0.5)
            continue 
        end
        
        local player = getPlayer()
        local character = getCharacter()
        local humanoid = getHumanoid()
        
        if not player or not character or not humanoid or humanoid.Health <= 0 then
            wait(1)
            continue
        end
        
        -- Check level
        local level = getLevel()
        if level >= config.TargetLevel then
            wait(5)
            continue
        end
        
        -- Auto Quest
        if config.AutoQuest then
            autoQuest()
        end
        
        -- Find target
        local target = nil
        if config.FarmBoss then
            target = findNearestBoss()
        end
        if not target then
            target = findNearestMonster()
        end
        
        -- Attack
        if target then
            fastAttack(target)
        else
            -- Move to spawn
            if config.TeleportToMonsters then
                local spawns = {
                    Vector3.new(1000, 0, 1000),
                    Vector3.new(-1000, 0, -1000),
                    Vector3.new(1500, 0, -500),
                    Vector3.new(-500, 0, 1500),
                    Vector3.new(0, 0, 2000),
                    Vector3.new(2000, 0, 0)
                }
                getRoot().CFrame = CFrame.new(spawns[math.random(1, #spawns)])
                wait(0.5)
            end
        end
        
        -- Auto Heal
        if config.AutoHeal then
            autoHeal()
        end
        
        -- Auto Collect
        if config.AutoCollect then
            autoCollect()
        end
        
        -- Server hop
        if config.ServerHop and config.HopWhenNoMonsters then
            local monsterCount = 0
            for _, v in pairs(game.Workspace:GetChildren()) do
                if v:IsA("Model") and v:FindFirstChild("Humanoid") and v.Name ~= player.Name then
                    monsterCount = monsterCount + 1
                end
            end
            if monsterCount < 5 then
                serverHop()
            end
        end
    end
end)()
